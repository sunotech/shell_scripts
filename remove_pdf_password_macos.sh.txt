#!/bin/bash

password=$1

# Get password from command line
if [[ -n $password ]]; then
    echo "password passed on"
fi

# Ask user for a password
if [[ -z $password ]]; then
    read -p "What is the password of the PDF(s)? " password
fi

# Exit if user doesn't provide a password
if [[ -z $password ]]; then
    echo "no password provided, exiting..."
    exit 1
fi

echo "starting..."

if ! command -v qpdf &> /dev/null; then
    echo "qpdf not found. Install it with 'brew install qpdf' and try again."
    exit 1
fi

mkdir -p orig

for i in *.[pP][dD][fF]; do
    if [[ -f "$i" ]]; then
        # Backup original
        cp "$i" orig/
        # Decrypt to temporary file
        qpdf --password="$password" --decrypt "$i" "${i}.tmp"
        # Replace original with decrypted version
        mv "${i}.tmp" "$i"
    fi
done

echo "done!"
